🔗 Google Apps Script Endpoint

วิธีการตั้งค่า:
1. เปิด Google Sheet → Extensions → Apps Script
2. วาง Code ของ Web App แล้วกด Deploy → New Deployment → Web App
3. ตั้ง Execute as: Me | Who has access: Anyone
4. คัดลอก Web App URL มาวางในช่องด้านล่าง
ดูได้จาก URL ของ Google Sheet ของท่าน

📋 Google Apps Script Code (สำหรับ Paste) 

// ===== วาง Code นี้ทั้งหมดใน Google Apps Script =====
const SHEET_ID   = ''; // ใส่ Spreadsheet ID ของท่าน
const SHEET_TABS = { personnel:'กำลังพล', promo:'เลื่อนยศ', courses:'หลักสูตร', ops:'ภารกิจ', docs:'เอกสาร' };

function doGet(e) {
  const p      = e.parameter;
  const cb     = p.callback || '';
  const action = p.action   || '';
  const table  = p.table    || 'personnel';
  let result;
  try {
    if      (action === 'getAll') result = getAllRows(table);
    else if (action === 'getOne') result = getOneRow(table, p.id);
    else result = { status:'error', message:'Unknown action' };
  } catch(err) { result = { status:'error', message: err.toString() }; }
  const json = JSON.stringify(result);
  return cb
    ? ContentService.createTextOutput(cb+'('+json+')').setMimeType(ContentService.MimeType.JAVASCRIPT)
    : ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  let data;
  try { data = JSON.parse(e.parameter.payload || e.postData.contents); }
  catch(err) { return out({ status:'error', message:'Parse error' }); }
  const action = data.action || '';
  const table  = data.table  || 'personnel';
  try {
    if      (action === 'add')    return out(addRow(table, data));
    else if (action === 'update') return out(updateRow(table, data));
    else if (action === 'delete') return out(deleteRow(table, data.id || data.ID));
    else return out({ status:'error', message:'Unknown action' });
  } catch(err) { return out({ status:'error', message: err.toString() }); }
}

function out(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

function getSheet(table) {
  const ss   = SpreadsheetApp.openById(SHEET_ID);
  const name = SHEET_TABS[table] || table;
  let sh = ss.getSheetByName(name);
  if (!sh) sh = ss.insertSheet(name);
  return sh;
}

function getAllRows(table) {
  const sh   = getSheet(table);
  const vals = sh.getDataRange().getValues();
  if (vals.length < 2) return { status:'ok', data:[] };
  const headers = vals[0];
  const data    = vals.slice(1).map(row => {
    const obj = {};
    headers.forEach((h, i) => { obj[h] = row[i] !== undefined ? String(row[i]) : ''; });
    return obj;
  }).filter(r => r['ID'] && r['ID'] !== '');
  return { status:'ok', data };
}

function getOneRow(table, id) {
  const all = getAllRows(table);
  const row = (all.data || []).find(r => r['ID'] === id);
  return row ? { status:'ok', data:row } : { status:'error', message:'Not found' };
}

function addRow(table, data) {
  const sh      = getSheet(table);
  const vals    = sh.getDataRange().getValues();
  const headers = vals.length > 0 ? vals[0] : Object.keys(data).filter(k => k !== 'action' && k !== 'table');
  if (vals.length === 0) sh.appendRow(headers);
  const newId = 'ID_' + Date.now();
  data['ID']  = newId;
  const row   = headers.map(h => data[h] !== undefined ? data[h] : '');
  sh.appendRow(row);
  return { status:'ok', id: newId };
}

function updateRow(table, data) {
  const sh      = getSheet(table);
  const vals    = sh.getDataRange().getValues();
  if (vals.length < 2) return { status:'error', message:'No data' };
  const headers = vals[0];
  const idCol   = headers.indexOf('ID');
  if (idCol < 0) return { status:'error', message:'No ID column' };
  for (let i = 1; i < vals.length; i++) {
    if (String(vals[i][idCol]) === String(data['ID'])) {
      headers.forEach((h, ci) => {
        if (h !== 'ID' && data[h] !== undefined)
          sh.getRange(i+1, ci+1).setValue(data[h]);
      });
      return { status:'ok' };
    }
  }
  return { status:'error', message:'Row not found' };
}

function deleteRow(table, id) {
  const sh      = getSheet(table);
  const vals    = sh.getDataRange().getValues();
  if (vals.length < 2) return { status:'error', message:'No data' };
  const idCol   = vals[0].indexOf('ID');
  for (let i = vals.length - 1; i >= 1; i--) {
    if (String(vals[i][idCol]) === String(id)) {
      sh.deleteRow(i + 1);
      return { status:'ok' };
    }
  }
  return { status:'error', message:'Row not found' };
}
