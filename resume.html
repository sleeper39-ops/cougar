📋 Google Apps Script Code (สำหรับ Paste) 

// ===== วาง Code นี้ทั้งหมดใน Google Apps Script =====
const SHEET_ID = ''; // ใส่ Spreadsheet ID ของท่านที่นี่ (หรือปล่อยว่างและตั้งค่าผ่าน Settings)

// ชื่อ Tab เริ่มต้น (ใช้เมื่อ client ไม่ส่ง tabName มา)
const DEFAULT_TABS = {
  personnel: 'กำลังพล',
  promo:     'เลื่อนยศ',
  courses:   'หลักสูตร',
  ops:       'ภารกิจ',
  docs:      'เอกสาร'
};

function doGet(e) {
  const p       = e.parameter;
  const cb      = p.callback || '';
  const action  = p.action   || '';
  const table   = p.table    || 'personnel';
  const tabName = p.tabName  || DEFAULT_TABS[table] || table; // รับชื่อ Tab จาก client
  let result;
  try {
    if      (action === 'getAll') result = getAllRows(tabName);
    else if (action === 'getOne') result = getOneRow(tabName, p.id);
    else result = { status:'error', message:'Unknown action' };
  } catch(err) { result = { status:'error', message: err.toString() }; }
  const json = JSON.stringify(result);
  return cb
    ? ContentService.createTextOutput(cb+'('+json+')').setMimeType(ContentService.MimeType.JAVASCRIPT)
    : ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  let data;
  try { data = JSON.parse(e.parameter.payload || e.postData.contents); }
  catch(err) { return out({ status:'error', message:'Parse error' }); }
  const action  = data.action  || '';
  const table   = data.table   || 'personnel';
  const tabName = data.tabName || DEFAULT_TABS[table] || table; // รับชื่อ Tab จาก client
  try {
    if      (action === 'add')    return out(addRow(tabName, data));
    else if (action === 'update') return out(updateRow(tabName, data));
    else if (action === 'delete') return out(deleteRow(tabName, data.id || data.ID));
    else return out({ status:'error', message:'Unknown action' });
  } catch(err) { return out({ status:'error', message: err.toString() }); }
}

function out(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

function getSheet(tabName) {
  const ssId = SHEET_ID || PropertiesService.getScriptProperties().getProperty('SHEET_ID') || '';
  const ss   = ssId ? SpreadsheetApp.openById(ssId) : SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(tabName);
  if (!sh) sh = ss.insertSheet(tabName); // สร้าง Tab ใหม่ถ้ายังไม่มี
  return sh;
}

function getAllRows(tabName) {
  const sh   = getSheet(tabName);
  const vals = sh.getDataRange().getValues();
  if (vals.length < 2) return { status:'ok', data:[] };
  const headers = vals[0];
  const data    = vals.slice(1).map(row => {
    const obj = {};
    headers.forEach((h, i) => { obj[h] = row[i] !== undefined ? String(row[i]) : ''; });
    return obj;
  }).filter(r => r['ID'] && r['ID'] !== '');
  return { status:'ok', data };
}

function getOneRow(tabName, id) {
  const all = getAllRows(tabName);
  const row = (all.data || []).find(r => r['ID'] === id);
  return row ? { status:'ok', data:row } : { status:'error', message:'Not found' };
}

function addRow(tabName, data) {
  const sh      = getSheet(tabName);
  const vals    = sh.getDataRange().getValues();
  const headers = vals.length > 0 ? vals[0] : Object.keys(data).filter(k => !['action','table','tabName'].includes(k));
  if (vals.length === 0) sh.appendRow(headers);
  const newId = 'ID_' + Date.now();
  data['ID']  = newId;
  const row   = headers.map(h => data[h] !== undefined ? data[h] : '');
  sh.appendRow(row);
  return { status:'ok', id: newId };
}

function updateRow(tabName, data) {
  const sh      = getSheet(tabName);
  const vals    = sh.getDataRange().getValues();
  if (vals.length < 2) return { status:'error', message:'No data' };
  const headers = vals[0];
  const idCol   = headers.indexOf('ID');
  if (idCol < 0) return { status:'error', message:'No ID column' };
  for (let i = 1; i < vals.length; i++) {
    if (String(vals[i][idCol]) === String(data['ID'])) {
      headers.forEach((h, ci) => {
        if (h !== 'ID' && data[h] !== undefined)
          sh.getRange(i+1, ci+1).setValue(data[h]);
      });
      return { status:'ok' };
    }
  }
  return { status:'error', message:'Row not found' };
}

function deleteRow(tabName, id) {
  const sh      = getSheet(tabName);
  const vals    = sh.getDataRange().getValues();
  if (vals.length < 2) return { status:'error', message:'No data' };
  const idCol   = vals[0].indexOf('ID');
  for (let i = vals.length - 1; i >= 1; i--) {
    if (String(vals[i][idCol]) === String(id)) {
      sh.deleteRow(i + 1);
      return { status:'ok' };
    }
  }
  return { status:'error', message:'Row not found' };
}

📋 Apps Script — เพิ่มเติมสำหรับ Google Drive

เพิ่ม Code นี้ต่อท้าย Apps Script เดิมในไฟล์เดียวกัน
// ===== เพิ่มใน doGet — รองรับ action drive =====
// (แทน doGet เดิมด้วยอันนี้)
function doGet(e) {
  const p       = e.parameter;
  const cb      = p.callback || '';
  const action  = p.action   || '';
  const table   = p.table    || 'personnel';
  const tabName = p.tabName  || DEFAULT_TABS[table] || table;
  let result;
  try {
    if      (action === 'getAll')       result = getAllRows(tabName);
    else if (action === 'getOne')       result = getOneRow(tabName, p.id);
    else if (action === 'driveTest')    result = testDriveFolder(p.folderId);
    else if (action === 'driveList')    result = listDriveFiles(p.folderId, p.prefix || '');
    else result = { status:'error', message:'Unknown action' };
  } catch(err) { result = { status:'error', message: err.toString() }; }
  const json = JSON.stringify(result);
  return cb
    ? ContentService.createTextOutput(cb+'('+json+')').setMimeType(ContentService.MimeType.JAVASCRIPT)
    : ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON);
}

// ===== เพิ่มใน doPost — รองรับ upload ไฟล์ =====
// (แทน doPost เดิมด้วยอันนี้)
function doPost(e) {
  let data;
  try { data = JSON.parse(e.parameter.payload || e.postData.contents); }
  catch(err) { return out({ status:'error', message:'Parse error' }); }
  const action  = data.action  || '';
  const table   = data.table   || 'personnel';
  const tabName = data.tabName || DEFAULT_TABS[table] || table;
  try {
    if      (action === 'add')          return out(addRow(tabName, data));
    else if (action === 'update')       return out(updateRow(tabName, data));
    else if (action === 'delete')       return out(deleteRow(tabName, data.id || data.ID));
    else if (action === 'uploadFile')   return out(uploadFileToDrive(data));
    else if (action === 'deleteFile')   return out(deleteFileFromDrive(data.fileId));
    else return out({ status:'error', message:'Unknown action' });
  } catch(err) { return out({ status:'error', message: err.toString() }); }
}

// ===== Google Drive Functions =====

// ทดสอบการเข้าถึง Folder
function testDriveFolder(folderId) {
  if (!folderId) return { status:'error', message:'ไม่ได้ระบุ Folder ID' };
  try {
    const folder = DriveApp.getFolderById(folderId);
    const name   = folder.getName();
    const files  = folder.getFiles();
    let count = 0;
    while (files.hasNext()) { files.next(); count++; }
    return { status:'ok', name, fileCount: count };
  } catch(e) {
    return { status:'error', message:'เข้าไม่ได้: ' + e.toString() };
  }
}

// ดูรายการไฟล์ใน Folder
function listDriveFiles(folderId, prefix) {
  if (!folderId) return { status:'error', message:'ไม่ได้ระบุ Folder ID' };
  try {
    const folder = DriveApp.getFolderById(folderId);
    const files  = folder.getFiles();
    const result = [];
    while (files.hasNext()) {
      const f = files.next();
      const name = f.getName();
      if (!prefix || name.startsWith(prefix)) {
        result.push({
          id:       f.getId(),
          name:     name,
          mimeType: f.getMimeType(),
          size:     f.getSize(),
          url:      f.getUrl(),
          viewUrl:  'https://drive.google.com/uc?id=' + f.getId(),
          modified: f.getLastUpdated().toISOString()
        });
      }
    }
    return { status:'ok', data: result };
  } catch(e) {
    return { status:'error', message: e.toString() };
  }
}

// อัปโหลดไฟล์ไปยัง Drive (รับ Base64)
function uploadFileToDrive(data) {
  const folderId  = data.folderId;
  const fileName  = data.fileName  || ('file_' + Date.now());
  const mimeType  = data.mimeType  || 'application/octet-stream';
  const base64    = data.base64;   // ข้อมูลไฟล์ในรูป base64
  if (!folderId || !base64) return { status:'error', message:'ขาด folderId หรือ base64' };
  try {
    const folder   = DriveApp.getFolderById(folderId);
    const decoded  = Utilities.base64Decode(base64);
    const blob     = Utilities.newBlob(decoded, mimeType, fileName);
    const file     = folder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    return {
      status:  'ok',
      fileId:  file.getId(),
      url:     file.getUrl(),
      viewUrl: 'https://drive.google.com/uc?id=' + file.getId(),
      name:    file.getName()
    };
  } catch(e) {
    return { status:'error', message: e.toString() };
  }
}

// ลบไฟล์จาก Drive
function deleteFileFromDrive(fileId) {
  if (!fileId) return { status:'error', message:'ไม่ได้ระบุ fileId' };
  try {
    DriveApp.getFileById(fileId).setTrashed(true);
    return { status:'ok' };
  } catch(e) {
    return { status:'error', message: e.toString() };
  }
}
